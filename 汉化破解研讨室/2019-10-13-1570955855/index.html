一个通用的 Galgame 游戏脚本修改工具

另附一文本库的例子……SDK啥的也懒的写了…研究下源码里"Library"文件夹就知道了，其实就是个接口= =

说说我的思想吧。
把 面对翻译的编辑器 和 面对程序的解析器 分开，文本的提取和封回 交由库处理（就是下面那段代码），编辑后前台把文本发送到库，库再进行处理写回到脚本中。这样，程序就不用再写可恶的GUI了……

有问题请回帖或PM，要更新嘛…基本不会有啥动作了，最多是把没完成的功能写完（因为原本就是练手作囧）

//以"//@Ref:"开头的行中冒号后的字符串将作为引用在编译时传递给编译器
//@Ref:System.dll
using System;
using System.Collections.Generic;
using System.Text;

namespace Text_Library
{
    class Program
    {
        #region Interface methods

        //原始文件内容
        string[] contents;
        //原始字符串
        string strEditableLines = String.Empty;
        //有效行列表
        List<int> lineNumber = new List<int>();

        //库信息
        public string DLLGetLibraryInfo()
        {
            return "Yuka Script";
        }

        //支持脚本的后缀
        public string DLLGetSupportExtensions()
        {
            return "*.txt|*.txt";
        }

        //得到在与前台传递文本时被认为是一句话结束的标记
        public string DLLGetStringBreakLineChar()
        {
            return Environment.NewLine;
        }

        //前台是否允许回车
        public bool DLLGetCanMakeNewLine()
        {
            return false;
        }

        //如果是，回车在传递回库时将被替换为……
        public string DLLGetNewLineBreakChar()
        {
            return @"\n";
        }

        //从前台打开的一个二进制脚本里获得可编辑的字符串表，字符串之间用DLLGetStringBreakLineChar()分隔
        //注意保存原始文件副本以便回写时使用
        public string DLLGetStringCollection(byte[] file)
        {
            string[] strLines = Encoding.Default.GetString(file).Replace(Environment.NewLine, "\n").Split('\n');
            //保存原始文件副本
            contents = strLines;

            int number = 0;
            foreach (string strLine in strLines)
            {
                if (strLine.Length < 7)
                {
                    number++;
                    continue;
                }

                if ((strLine.StartsWith("STR ")) && (!strLine.Contains("\\")))
                {
                    lineNumber.Add(number);
                    strEditableLines += strLine.Substring(5, strLine.LastIndexOf('\"') - 5) + Environment.NewLine;
                }

                number++;
            }

            //去除末尾的换行
            strEditableLines = strEditableLines.Substring(0, strEditableLines.Length - Environment.NewLine.Length);

            return strEditableLines;
        }

        //回写，传入字符串同上。返回值是新的脚本文件
        public byte[] DLLSetStringCollection(string str)
        {
            string[] strOrgLines = strEditableLines.Replace(DLLGetStringBreakLineChar(), "\n").Split('\n');
            string[] strNewLines = str.Replace(DLLGetStringBreakLineChar(), "\n").Split('\n');

            int number = 0;
            foreach (string line in strNewLines)
            {
                contents[lineNumber[number]] = contents[lineNumber[number]].Replace(strOrgLines[number], line);
                number++;
            }

            string sss = string.Empty;
            foreach (string s in contents)
                sss += s + Environment.NewLine;

            sss = sss.Substring(0, sss.Length - Environment.NewLine.Length);

            return Encoding.Default.GetBytes(sss);
        }

        #endregion Interface methods
    }
}

一直在做这方面的实验。旧版本中从脚本中读取一行文本（是的，旧版本不支持二进制，只支持中间格式）交由DLL，DLL用某种匹配方式提取文本以后返回给编辑器，例如：
STR "Hello, World"   ----->   Hello, World
在新版中，为了支持二进制脚本，编辑器的工作就仅仅是读取二进制脚本内容传递给DLL，DLL内部处理后直接把可编辑的文本返回给编辑器，这样的好处就是能更加灵活的控制返回的行（例如可以返回注释行，“【注释】语音：阿雅”），缺点就是在DLL中要手写各种解析代码Orz，为了尽量减少工作…新版用了DLL动态调用，也就是DLL中声明为static的变量会在编辑器打开的整个时期保留…这样会造成内存占用增大，不过现在都是2G了，多个20M也没关系吧……
