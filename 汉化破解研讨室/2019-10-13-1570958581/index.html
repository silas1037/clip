typedef unsigned char BYTE;

int Decompress(int v, BYTE *inp, BYTE *oup) {
    const int windowSize=0x1000, windowMask=0xFFF;
    BYTE window[windowSize];    // Circular queue that storage the most recent windowSize bytes of decompressed data
    int windowPointer=0;

    BYTE bitmap=0, mask=0;
    BYTE *initOup=oup;

    int size=*(int*)inp;
    inp+=4;

    while(size) {
        if (!(mask>>=1)) {    //bitmap used up. read new one.
            bitmap=*inp++;
            size--;
            if (size==0) break;
            mask=0x80;
        }

        if (bitmap&mask) {
            //1 byte of direct data
            window[windowPointer++]=*inp++;
            size--;
            if (windowPointer>=windowSize) {    //flush buff
                BYTE *b=window;
                while (windowPointer--) *oup++=*b++;
            }
        } else {
            // dictionary data
            if (size>=2) {
                BYTE firstByte=*inp++;
                BYTE secondByte=*inp++;
                size-=2;

                int offset=(firstByte&0xf0)<<4+secondByte;
                int prev=windowPointer-offset;
                int len=(firstByte&0xf)+3;

                // 'offset' bytes before, the same string appears 'len' times. Fill into buff
                while (len--) {
                    window[windowPointer++]=window[(prev++)&windowMask];
                    if (windowPointer>=windowSize) {    // flush buff if it is full
                        BYTE *b=window;
                        while (windowPointer--) *oup++=*b++;
                    }
                }
            } else break;
        }
    }

    while(windowPointer--) {    // flush buff
        BYTE *b=window;
        *oup++=*b++;
    }
    return oup-initOup;    // length of decompressed data
}

typedef unsigned char BYTE;

int Decompress(int v, BYTE *inp, BYTE *oup) {
    BYTE bitmap=0, mask=0, *initOup=oup, *end=inp+4+*(int*)inp;
    inp+=4;
    while(inp<end) {
        if (!(mask>>=1)) {
            bitmap=*inp++;
            if (inp>=end) break;
            mask=0x80;
        }
        if (bitmap&mask) *oup++=*inp++;

            int len=*inp++;

            len=(len&0xF)+3;
            while(len--) *oup++=*(oup-offset);    // may underflow here
        }
    }
    return oup-initOup;
}
