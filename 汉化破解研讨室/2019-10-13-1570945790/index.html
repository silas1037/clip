自己试着跟踪了一下游戏，系统为“AdvSystem”
分析出来一部分.sc格式：

header：
1字节标识符        0xFF
4字节整数        在文件尾部有一段列表，这是表的索引数，用途未知
4字节整数        代码段长度（接在header之后）
4字节整数        资源段长度（接在代码段之后）

代码段：
始终以 1E 00 00 00 开头
似乎用了代号来替换指令
05 00 00 00 FF FF FF FF 之后跟着的4字节整数代表在资源段所需资源的偏移地址（从资源段开头算起）

资源段：
整个资源段做了“非”加密，解密后就是标准字符串格式：字符串+0x00结尾
还有就是代码段里的引用量也当作资源处理，和文本堆在一起，不好分离
接着又发现有的字符串间居然有两个0x00，不知道意义何在，要具体测试后才能知道

未知指令段：
4字节整数        指令数量，如果等于0，那么到此为止
    1字节整数    指令名长度
    指令名长度    指令名
    4字节整数    这4字节整数都可以在代码段内找到并重复多次，疑为指令代号
依此反复该结构[指令数量]次

未知表：
4字节整数        索引数，与header段里的索引数相等
    4字节整数    索引值，从0开始一直到 [索引数]-1 共[索引数]个
    4字节整数    单调增加的值，用途未知

最后还有个不明用途的字符串：
4字节整数        只有两个值，scenario.sc是0x0B，其它是0x02
4字节整数        0x09，所有脚本都是这个值
之后是字符串，之间有“00 00 00”，但不完全如此，有的之间插进了个0x09，所以格式不明……

自己没法把所有的命令的功能都给解析出来……但是要做汉化文本替换的话，这些信息应该够

自己写了一个分离器：
EASC.py
Quote:
# For Windows OS Only....

import sys
import os
import struct

Version="0.01"

if len(sys.argv) < 2:
  print "Easy AdvSys SC Dis/Assembler ver "+Version+"\n"
  print "Usage: "+sys.argv[0]+" <.SCFile or .KONFile>"
  quit()

# Debug:
#sys.argv.append("D:\\Moshi\\scenario.sc")

PATH = sys.path[0]
if not os.path.isdir(PATH):
  PATH=os.path.dirname(PATH)

def ExtractSC():
  if not os.path.isfile(sys.argv[1]):
    print "No this file."
    quit()
  SCFile=open(sys.argv[1],'rb')
  if not SCFile.read(1)==Magic:
    print "Bad SC Format."
    quit()
  IndexLength,CodeLength,ResLength=struct.unpack('iii',SCFile.read(12))
  CodeLength=CodeLength/4
  Code=[]
  CodeTable={}
  for n in range(CodeLength):
    Code.append(struct.unpack('i',SCFile.read(4))[0])
  StringDic={}
  ResString=''.join([chr(ord(x)^0xff) for x in SCFile.read(ResLength)])
  ResIndex={}
  idx=0
  offset=0
  while offset<ResLength:
    while not ResString[offset]=='\0':
      tmp=ResString.find('\0',offset)
      StringDic[idx]=ResString[offset:tmp]
      ResIndex[offset]=idx
      offset=tmp
      idx+=1
    offset+=1
  CommandNum=struct.unpack('i',SCFile.read(4))[0]
  CommandName=[]
  CommandCode=[]
  for n in range(CommandNum):
    CommandNameLen=ord(SCFile.read(1))
    CommandName.append(SCFile.read(CommandNameLen))
    CommandCode.append(struct.unpack('i',SCFile.read(4))[0])
    CodeTable[CommandCode[n]]=CommandName[n]
  IndexLength=struct.unpack('i',SCFile.read(4))[0]
  TableIndex=[]
  TableValue=[]
  for n in range(IndexLength):
    IndexN,ValueN=struct.unpack('ii',SCFile.read(8))
    TableIndex.append(IndexN)
    TableValue.append(ValueN)
  JunkString=SCFile.read()
  SCFile.close()
  ResFile=open(sys.argv[1][:-3]+'.RES','w')
  ResFile.write("#AdvSystem ResourceFile Created by EASC Ver "+Version+'\n\n')
  for n in range(len(ResIndex)):
    temp='<%05d> ' % n
    ResFile.write(temp+StringDic[n].replace("\x0a",r"\n",6)+'\n')
  ResFile.close()
  CodeFile=open(sys.argv[1][:-3]+'.KON','w')
  CodeFile.write("#AdvSystem ControlScript Created by EASC Ver "+Version+'\n\n')
  IsRes=0
  n=0
  while n<CodeLength:
    if Code[n]==-1:
      CodeFile.write('Res')
      IsRes=True
      n+=1
      continue
    if IsRes:
      if ResIndex.has_key(Code[n]):
        CodeFile.write('<%05d>\n' % ResIndex[Code[n]])
      else:
        CodeFile.write('<%05d>\n' % ResIndex[Code[n]+1])
      IsRes=False
      n+=1
      continue
    if Code[n]==1 and Code[n+1]==1:
      CodeFile.write('S11\n')
      n+=2
      continue
    if Code[n]==2 and Code[n+1]==1:
      CodeFile.write('S21\n')
      n+=2
      continue
    if Code[n]==1:
      CodeFile.write('Use')
      IsRes=True
      n+=1
      continue
    if CodeTable.get(Code[n],0)==0:
      CodeFile.write(str(Code[n])+'\n')
      n+=1
      continue
    CodeFile.write(CodeTable[Code[n]]+'\n')
    n+=1
  CodeFile.close()
  TableFile=open(sys.argv[1][:-3]+'.TBF','w')
  TableFile.write("#AdvSystem TableMap Created by EASC Ver "+Version+'\n\n')
  TableFile.write("$CodeMap\n")
  for n in range(CommandNum):
    TableFile.write(str(CommandCode[n]).ljust(11)+CommandName[n]+'\n')
  TableFile.write("\n$UnknownTable\n")
  for n in range(IndexLength):
    TableFile.write(str(TableIndex[n]).ljust(8)+str(TableValue[n])+'\n')
  TableFile.close()
  JunkFile=open(sys.argv[1][:-3]+'.UKN','wb')
  JunkFile.write(JunkString)
  JunkFile.close()
  print "Finished."
  quit()

def PackSC():
  FileName=sys.argv[1][:-4]
  if not os.path.isfile(FileName+".res") and os.path.isfile(sys.argv[1]) and os.path.isfile(FileName+".tbf") and os.path.isfile(FileName+".ukn"):
    print "File lost..."
    quit()
  KONFile=open(sys.argv[1],'r')
  RESFile=open(FileName+".res",'r')
  TBFFile=open(FileName+".tbf",'r')
  RESDict={}
  RESOffset={}
  CommandDict={}
  CommandNotUsed={}
  CommandNum=0
  RESString=''
  KONString=''
  CommandString=''
  Table=[]
  ReadString=RESFile.readline().split(" ",1)
  while ReadString[0]:
    if len(ReadString)>1 and ReadString[0][0]=='<':
      RESDict[ReadString[0][1:6]]=ReadString[1][:-1]
    ReadString=RESFile.readline().split(" ",1)
  ReadString=TBFFile.readline()
  while not ReadString=="$CodeMap\n":
    ReadString=TBFFile.readline()
  ReadString=TBFFile.readline().split(" ",1)
  while not ReadString[0]=='\n':
    CommandDict[ReadString[1].strip(" \n")]=int(ReadString[0])
    CommandNotUsed[ReadString[1].strip(" \n")]=False
    ReadString=TBFFile.readline().split(" ",1)
  while not ReadString=="$UnknownTable\n":
    ReadString=TBFFile.readline()
  ReadString=TBFFile.readline().split(" ",1)
  while len(ReadString)==2:
    Table.append(int(ReadString[1].strip("\n")))
    ReadString=TBFFile.readline().split(" ",1)
  idx=0
  RESLength=len(RESDict)
  while idx<RESLength:
    RESOffset['%05d' % idx]=len(RESString)
    RESString+=RESDict['%05d' % idx].replace('\\n','\n',6)+'\0'
    idx+=1
  ReadString=KONFile.readline()
  ReadString=KONFile.readline()
  ReadString=KONFile.readline().strip("\n")
  while ReadString:
    if ReadString[0:3]=="S21":
      KONString+='\2\0\0\0\1\0\0\0'
      ReadString=KONFile.readline().strip("\n")
      continue
    if ReadString[0:3]=="Use":
      KONString+='\1\0\0\0'
      KONString+=struct.pack('i',RESOffset[ReadString[4:9]])
      ReadString=KONFile.readline().strip("\n")
      continue
    if ReadString[0:3]=="Res":
      KONString+='\xff\xff\xff\xff'
      KONString+=struct.pack('i',RESOffset[ReadString[4:9]])
      ReadString=KONFile.readline().strip("\n")
      continue
    if CommandDict.has_key(ReadString):
      if CommandNotUsed[ReadString]:
        CommandString+=struct.pack('b',len(ReadString))+ReadString+struct.pack('i',CommandDict[ReadString])
        CommandNum+=1
      KONString+=struct.pack('i',CommandDict[ReadString])
      ReadString=KONFile.readline().strip("\n")
      continue
    if ReadString[0:3]=="S11":
      KONString+='\1\0\0\0\1\0\0\0'
      ReadString=KONFile.readline().strip("\n")
      continue
    KONString+=struct.pack('i',int(ReadString))
    ReadString=KONFile.readline().strip("\n")
  RESFile.close()
  TBFFile.close()
  KONFile.close()
  SCFile=open(FileName+".SC",'wb')
  SCFile.write(Magic+struct.pack('iii',len(Table),len(KONString),len(RESString)))
  SCFile.write(KONString)
  SCFile.write(''.join([chr(ord(x)^0xff) for x in RESString]))
  SCFile.write(struct.pack('i',CommandNum))
  SCFile.write(CommandString)
  SCFile.write(struct.pack('i',len(Table)))
  for n in range(len(Table)):
    SCFile.write(struct.pack('ii',n,Table[n]))
  SCFile.write(open(FileName+".ukn",'rb').read())
  SCFile.close()
  print FileName+".SC Packed."
  quit()

Magic='\xFF'
if os.path.splitext(sys.argv[1])[1].lower()==".sc": ExtractSC()
if os.path.splitext(sys.argv[1])[1].lower()==".kon": PackSC()
print "Nothing to do with "+sys.argv[1]+"..."
quit()


会解出4个文件：.kon是控制代码，未解明的代码全都是数字……不要管它
.res是资源文件，jis->gbk以后修改其中的文字即可
.tbf是映射表（推测），不用管
.ukn，是unknow，直接就原样输出了，莫动

现在只要找个略懂日语的苦力把翻译档内容替换res文件了……

这里用的是猜测代码的含义方式来替换文本，但是往往会出现各种问题
所以还有B方案：即翻译文本100%比原文本短的话，以填充“00”的方式原封不动地按原始格式替换字符串
这种方法兼容性更佳，实现也不难……
