特别说明 这个未测试
因为作者不具备测试条件
导出了三个函数
其中RLE和GLZSS是Delphi专用
function Get_Pack_Data;
则是与c#等通用的函数
不过只能用于处理E2文件(即解析出来的图像文件)
直接修改自我的HALTE的插件
以下给出结构(我懒得修改了)

    TPack_Header = packed record    //文件头信息
      Magic : PChar;               //Magic
      Entries_Count : Cardinal;       //entries数量
      Entries_Offset  : Cardinal;     //entries起始地址
      Entries_Length_O  : Cardinal;   //初始entries长度
      Entries_Length  : Cardinal;     //备用
      Entries_Origin  : Cardinal;     //起始位置(0 = 文件头,2=文件末尾)
      Data_Offset : Cardinal;         //数据起始地址
      Data_Length_O : Cardinal;       //初始数据长度
      Data_Length : Cardinal;         //备用
      Data_Origin  : Cardinal;        //起始位置(0 = 文件头,2=文件末尾,3 = 数据位于其他文件中)
      File_Address  : PChar;       //当Data_Origin=3时,这里必须给出文件相对于Entries所在文件的地址
      Entries_Encoded : Cardinal;   //entries是否加密
      Data_Encoded : Cardinal;      //数据是否加密
      Additional_Data : Pointer;   //附加信息(可以根据需要储存密匙等必要信息),如果密匙储存于独立文件中,请将此pointer置为 Need_KeyFile (必须为WideString)
      Additional_Data_Length : Cardinal;  //附加信息长度,长度必须真实
      Additional_Entries : Cardinal;   //如果entries有多个部分,那么这里应当指明(只需要给出附加的entries数量即可)
      Entries_Array : TMy_EntriesArray;  //如果Additional_Entries不为0,此结构必须填充
      Err_Code  : Err_Code;             //错误码
    end;


TPack_Entries  = packed record    //Entry信息,没有errcode,出错时直接返回Nil即可.
      Offset  : Cardinal;          //偏移(相对于$00)
      Length  : Cardinal;          //长度
      Decompressed_Length : Cardinal; //对应压缩数据,根据需要填充(0:数据未压缩 $FFFFFFFF数据压缩但是未指明解压大小)
      Name  : PChar;            //文件名(必须符合win32标准,文件名最前端不能有'\',字符串必须顶格储存)
    end;
不需要填充这两个结构,自然留空即可
返回值不必关注,没有意义
output必须事先分配好空间
有BUG请告诉我...我相信一定有BUG

library Decompress;

{ Important note about DLL memory management: ShareMem must be the
  first unit in your library's USES clause AND your project's (select
  Project-View Source) USES clause if your DLL exports any procedures or
  functions that pass strings as parameters or function results. This
  applies to all strings passed to and from your DLL--even those that
  are nested in records and classes. ShareMem is the interface unit to
  the BORLNDMM.DLL shared memory manager, which must be deployed along
  with your DLL. To avoid using BORLNDMM.DLL, pass string information
  using PChar or ShortString parameters. }

uses
  SysUtils,
  windows,
  Classes,
  MyRec in 'MyRec.pas',
  ZLib,
  Generic_LZXX in 'Generic_LZXX.pas',
  AnimED_Math in 'AnimED_Math.pas';

{$R *.res}
const
Oggs = AnsiString('OggS');
Script = AnsiString('E1');

type
scr_head = packed record
   Magic : array[0..1] of AnsiChar;
   Data_Length : Cardinal;
   Decompress : Cardinal;
end;

procedure  RLE_Decompress(InputStream, OutputStream : TStream);
var
  cp: Cardinal;
  flag ,dat: Byte;
  i:Integer;
begin
  while True do
  begin
    InputStream.Read(flag,2);
    if flag = $ff then Break;
    InputStream.Read(cp,4);
    if flag =3 then
    begin
      InputStream.Read(dat,2);
      for I := 0 to cp - 1 do
      OutputStream.Write(dat,2);
    end else
    begin
      OutputStream.CopyFrom(InputStream,cp);
    end;
  end;
end;

function Get_Pack_Data(input:Pointer;input_length:Cardinal;output : Pointer ;Head:TPack_Header;Entry_Info:TPack_Entries):TData_Result;stdcall;
var
Com,decom:TmemoryStream;
Buffer:PByte;
Sc_head : scr_head;
Catscence : AnsiString;
begin
Catscence :=  Script;
if CompareMem(input,Pbyte(Catscence),2) then
begin
  Buffer := input;
  Move(Buffer^,Sc_head.Magic[0],$10);
  Entry_Info.Decompressed_Length := Sc_head.Decompress;
  Inc(Buffer,$10);
  decom:=Tmemorystream.create;
  Com:=Tmemorystream.create;
  com.Write(Buffer^,input_length - $10);
  com.Position:=0;
  decom.Position:=0;
  GLZSSDecode(com,decom,input_length - $10,$FEE,$FFF);
  if decom.Size <> entry_Info.Decompressed_Length then
  begin
    result.Err_Code.Num := Compress_Error;
    result.Err_Code.Info := Compress_Error_S;
    Exit;
  end;
  decom.Position:=5;
  Com.Clear;
  Com.Position :=0;
  RLE_Decompress(decom,Com);
  Com.Position :=0;
  com.Read(output^,com.Size);
  decom.Destroy;
  Result.Data := Buffer;
  Result.Length :=  com.Size;
  com.Destroy;
  Result.Err_Code.Num := 0;
  Result.Err_Code.Info := nil;
  Exit;
end else
begin
Result.Data := input;
Result.Length :=  input_length;
Result.Err_Code.Num := 0;
Result.Err_Code.Info := nil;
Exit;
end;
end;

exports
   Get_Pack_Data;
   exports
   RLE_Decompress;
   exports
   GLZSSDecode;

begin
end.

TPack_Header = packed record    //文件头信息
      Magic : PChar;               //Magic
      Entries_Count : Cardinal;       //entries数量
      Entries_Offset  : Cardinal;     //entries起始地址
      Entries_Length_O  : Cardinal;   //初始entries长度
      Entries_Length  : Cardinal;     //备用
      Entries_Origin  : Cardinal;     //起始位置(0 = 文件头,2=文件末尾)
      Data_Offset : Cardinal;         //数据起始地址
      Data_Length_O : Cardinal;       //初始数据长度
      Data_Length : Cardinal;         //备用
      Data_Origin  : Cardinal;        //起始位置(0 = 文件头,2=文件末尾,3 = 数据位于其他文件中)
      File_Address  : PChar;       //当Data_Origin=3时,这里必须给出文件相对于Entries所在文件的地址
      Entries_Encoded : Cardinal;   //entries是否加密
      Data_Encoded : Cardinal;      //数据是否加密
      Additional_Data : Pointer;   //附加信息(可以根据需要储存密匙等必要信息),如果密匙储存于独立文件中,请将此pointer置为 Need_KeyFile (必须为WideString)
      Additional_Data_Length : Cardinal;  //附加信息长度,长度必须真实
      Additional_Entries : Cardinal;   //如果entries有多个部分,那么这里应当指明(只需要给出附加的entries数量即可)
      Entries_Array : TMy_EntriesArray;  //如果Additional_Entries不为0,此结构必须填充
      Err_Code  : Err_Code;             //错误码
    end;

      Name  : PChar;            //文件名(必须符合win32标准,文件名最前端不能有'\',字符串必须顶格储存)
    end;
