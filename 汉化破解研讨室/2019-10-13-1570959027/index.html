 求助CIRCUS社引擎,PCM压缩音频解压后转换WAV无法播放

//////////////////////////////////////////////////////////////////////////////////////////////////////////

#include <windows.h>
#include <stdio.h>
#include "zlib123/zlib.h"
#pragma comment(lib,"zlib1.lib")

typedef struct {
    char magic[4];          /* XPCM */
    unsigned long length; /* length(需要重新计算) */
    byte mode;              /* 0 - raw , 3 - zlib */
    byte reserved[3];
    unsigned short wFormatTag;
    unsigned short nChannels;
    unsigned long nSamplesPerSec;
    unsigned long nAvgBytesPerSec;
    unsigned short nBlockAlign;
    unsigned short wBitsPerSample;
    unsigned long compr_length;/*mode=0时无效*/
    /* BYTE *compr_data */
} pcm_header_t;

unsigned long CRP_MakePCM3Length(unsigned long len){
    
    unsigned long rnt=0;
    _asm{
               xor eax,eax
            mov eax,0x81020409
            mov ebx,len
            imul ebx
            add edx,ebx
            sar edx,0x0B
            mov eax,edx
            shr eax,0x1F
            add eax,edx
            shl eax,0x0C
            add eax,0x4002
            mov rnt,eax
    }
    
    return rnt;
    
}


void CRP_ExtractPCM(const char* filename){

    byte* uncmp=NULL;
    unsigned long uncmp_length=0;
    unsigned long dwWritten=0;
    byte* lpdata=NULL;
    pcm_header_t pcm_header;
    HANDLE fr=CreateFile(filename,GENERIC_READ,FILE_SHARE_READ,NULL,OPEN_EXISTING,FILE_ATTRIBUTE_ARCHIVE,NULL);
    memset(&pcm_header,0,sizeof(pcm_header_t));
    ReadFile(fr,&pcm_header,sizeof(pcm_header_t),&dwWritten,NULL);
    if (pcm_header.mode==3)//zlib
    {

        uncmp_length=CRP_MakePCM3Length(pcm_header.length);
        SetFilePointer(fr,sizeof(pcm_header_t),NULL,FILE_BEGIN);
        lpdata=new byte[pcm_header.compr_length];
        ReadFile(fr,lpdata,pcm_header.compr_length,&dwWritten,NULL);
        uncmp = new byte[uncmp_length];
        if(uncompress((BYTE *)uncmp, &uncmp_length,(BYTE *)lpdata,pcm_header.compr_length))printf("error!~\n");
        delete[]lpdata;

        //
        pcm_header.length=uncmp_length;

    }else
    if (pcm_header.mode==0)//no compress
    {
        SetFilePointer(fr,sizeof(pcm_header_t)-sizeof(unsigned long),NULL,FILE_BEGIN);
        uncmp=new byte[pcm_header.length];
        ReadFile(fr,uncmp,pcm_header.length,&dwWritten,NULL);
    }
    CloseHandle(fr);

    if (pcm_header.length&&uncmp)
    {
        WAVEFORMATEX wav_header;
        DWORD wav_length, riff_chunk_len, fmt_chunk_len, data_chunk_len;
        BYTE *wav;
        
        wav_header.wFormatTag = pcm_header.wFormatTag;
        wav_header.nChannels = pcm_header.nChannels;
        wav_header.wBitsPerSample = pcm_header.wBitsPerSample;
        wav_header.nSamplesPerSec = pcm_header.nSamplesPerSec;
        wav_header.cbSize = 0;
        wav_header.nBlockAlign = /*pcm_header.nChannels * pcm_header.wBitsPerSample / 8*/pcm_header.nBlockAlign;
        wav_header.nAvgBytesPerSec = /*wav_header.nBlockAlign * pcm_header.nSamplesPerSec*/pcm_header.nAvgBytesPerSec;
        wav_header.cbSize = 0;
        
        fmt_chunk_len = sizeof(WAVEFORMATEX) + 0;
        data_chunk_len = pcm_header.length;
        riff_chunk_len = (4) + (8 + fmt_chunk_len) + (8 + data_chunk_len);
        wav_length = 8 + riff_chunk_len;
        wav = (BYTE *)(malloc)(wav_length);
        
        memcpy(wav, "RIFF", 4);
        *(unsigned long *)&wav[4] = riff_chunk_len;
        memcpy(&wav[8], "WAVE", 4);
        memcpy(&wav[12], "fmt ", 4);
        *(unsigned long *)&wav[16] = fmt_chunk_len;
        memcpy(&wav[20], &wav_header, sizeof(wav_header));
        memcpy(&wav[38], NULL, 0);
        memcpy(&wav[38 + 0], "data", 4);
        *(unsigned long *)&wav[38 + 0 + 4] = data_chunk_len;
        memcpy(&wav[38 + 0 + 8], uncmp, data_chunk_len);
        
    //
        delete[]uncmp;

        HANDLE fw=NULL;
         char new_wav[MAX_PATH]={0};
         strncpy(new_wav,filename,strlen(filename)-4);
         strcat(new_wav,".wav");
         fw=CreateFile(new_wav,GENERIC_WRITE,FILE_SHARE_WRITE,NULL,CREATE_ALWAYS,FILE_ATTRIBUTE_ARCHIVE,NULL);
          WriteFile(fw,wav,wav_length,&dwWritten,NULL);
        free(wav);
         CloseHandle(fw);
    }
    
}


int main(int argc, char *argv[]){
CRP_ExtractPCM("test.pcm");
  return 0;
}
