procedure Decode(const Buffer: Pbyte; const length: dword);
var
  esi, i: Integer;
begin
  esi := 0;
  for i := 0 to length - 1 do
  begin
    Buffer := Buffer xor decode_table[esi];
    inc(esi);
    esi := esi and $800000FF;
  end;
end;

procedure _Decompress(Input, output: Tmemorystream);
var
  raw: byte;
  Wraw, wraw2: word;
  i, current, j: Integer;
  buff: byte;
begin
  Input.Position := 0;
  output.Position := 0;
  while True do
  begin
    Input.read(raw, 1);
    for i := 0 to 8 - 1 do
    begin
      if Input.Position = Input.Size then
        exit;
      if (raw and $01) = 1 then
        output.CopyFrom(Input, 1)
      else
      begin
        Input.read(Wraw, 2);
        wraw2 := Wraw;
        Wraw := Wraw shr 4;
        wraw2 := (wraw2 and $0F) + 2;
        current := output.Position;
        for j := 0 to wraw2 - 1 do
        begin
          output.Seek(current - Wraw, sofrombeginning);
          output.read(buff, 1);
          output.Seek(current, sofrombeginning);
          output.Write(buff, 1);
          inc(current);
        end;
      end;
      raw := raw shr 1;
    end;
  end;
end;

procedure DBG_WRITETOFILE(Input: Tmemorystream; filename: string);
var
  Tmp: Integer;
  Buffer: Pbyte;
  pos: Int64;
begin
  pos := Input.Position;
  Input.Position := 0;
  Getmem(Buffer, Input.Size);
  Input.read(Buffer[0], Input.Size);
  Tmp := fileopen(filename + '.tmp', fmopenwrite);
  if Tmp = -1 then
    Tmp := filecreate(filename + '.tmp');
  filewrite(Tmp, Buffer[0], Input.Size);
  fileclose(Tmp);
  Freemem(Buffer);
  Input.Position := pos;
end;

Function PAC_Ana_Res(str_File_Path: String): Boolean;
Var
  hdl_File_Handle: Integer;
  Head: Pack_Head2;
  OP: array of _op;
  TempBuffer: Pbyte;
  Int_vnCounter: Integer;
  Out_Handle: Integer;
  Compress, decompress: Tmemorystream;
  de_Length: dword;
begin
  Result := True;
  try
    hdl_File_Handle := fileopen(str_File_Path, fmopenread);
    if hdl_File_Handle = -1 then
      raise Exception.Create('Can Not Open File');
    FileRead(hdl_File_Handle, Head.Header_Length, $5C);
    setlength(OP, Head.text_file_num);
    fileseek(hdl_File_Handle, Head.text_file_offset, 0);
    FileRead(hdl_File_Handle, OP[0].offset, Head.text_file_num * 8);
    str_File_Path := ExtractFilePath(str_File_Path) + 'OUTPUT\' +
      ChangeFileExt(extractfilename(str_File_Path), '') + '\';
    for Int_vnCounter := 0 to Head.text_file_num - 1 do
    begin
      if not DirectoryExists
        (ExtractFilePath(str_File_Path + inttostr(Int_vnCounter))) then
        ForceDirectories(ExtractFilePath(str_File_Path +
          inttostr(Int_vnCounter)));
      Out_Handle := filecreate(str_File_Path + inttostr(Int_vnCounter));
      if Out_Handle = -1 then
        Out_Handle := fileopen(str_File_Path + inttostr(Int_vnCounter),
          fmopenwrite);
      TempBuffer := sysgetmem(OP[Int_vnCounter].length);
      fileseek(hdl_File_Handle, OP[Int_vnCounter].offset + Head.text_offset, 0);
      FileRead(hdl_File_Handle, TempBuffer[0], OP[Int_vnCounter].length);
      begin
        Decode(TempBuffer, OP[Int_vnCounter].length);
        move(TempBuffer[4], de_Length, 4);
        decompress := Tmemorystream.Create;
        Compress := Tmemorystream.Create;
        Compress.Write(TempBuffer[8], OP[Int_vnCounter].length - 8);
        Compress.Position := 0;
        decompress.Position := 0;
        _Decompress(Compress, decompress);
        Compress.Destroy;
        sysfreemem(TempBuffer);
        TempBuffer := sysgetmem(decompress.Size);
        decompress.Position := 0;
        decompress.read(TempBuffer[0], decompress.Size);
        OP[Int_vnCounter].length := decompress.Size;
        decompress.Destroy;
      end;
      filewrite(Out_Handle, TempBuffer[0], OP[Int_vnCounter].length);
      sysfreemem(TempBuffer);
      fileclose(Out_Handle);
    end;
    fileclose(hdl_File_Handle);
  except
    Result := False;
    exit;
  end;
end;

Function PAC_Ana_Sce(str_File_Path: String): Boolean;
var
  Head: Head_Scr;
  Buffer: Tmemorystream;
  i: Integer;
  OP: array of _op;
  Tstr: string;
  OBuffer: Tstringlist;
begin
  Result := True;
  try
    Buffer := Tmemorystream.Create;
    Buffer.Clear;
    OBuffer := Tstringlist.Create;
    OBuffer.Clear;
    Buffer.LoadFromFile(str_File_Path);
    Buffer.read(Head.head_length, $84);
    setlength(OP, Head.string_table_num);
    Buffer.read(OP[0].offset, Head.string_table_num * 8);
    for i := 0 to Head.string_table_num do
    begin
      Buffer.Seek(OP.offset * 2 + Head.raw_string_offset, sofrombeginning);
      setlength(Tstr, OP.length);
      Buffer.read(Pbyte(Tstr)^, OP.length * 2);
      OBuffer.Add(Tstr);
    end;
    OBuffer.SaveToFile(str_File_Path + '.txt');
    OBuffer.Destroy;
    Buffer.Destroy;
  except
    Result := False;
    exit;
  end;
end;

function MakeFileList(Path, FileExt: string): Tstringlist;
var
  sch: TSearchrec;
begin
  Result := Tstringlist.Create;
  if rightStr(trim(Path), 1) <> '\' then
    Path := trim(Path) + '\'
  else
    Path := trim(Path);
  if not DirectoryExists(Path) then
  begin
    Result.Clear;
    exit;
  end;
  if FindFirst(Path + '*', faAnyfile, sch) = 0 then
  begin
    repeat
      Application.ProcessMessages;
      if ((sch.Name = '.') or (sch.Name = '..')) then
        Continue;
      if DirectoryExists(Path + sch.Name) then
      begin
        Result.AddStrings(MakeFileList(Path + sch.Name, FileExt));
      end
      else
      begin
        if (UpperCase(extractfileext(Path + sch.Name)) = UpperCase(FileExt)) or
          (FileExt = '.*') then
          Result.Add(Path + sch.Name);
      end;
    until FindNext(sch) <> 0;
    SysUtils.FindClose(sch);
  end;
end;
