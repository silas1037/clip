procedure ZlibDecompressStream(Source,Dest:TMemoryStream;DestLen:LongWord);
var
  zstream: TZStreamRec;
  SourceLen:LongWord;
begin
  FillChar(zstream,SizeOf(TZStreamRec),0);
  SourceLen:=Source.Size;
  Source.Position:=0;
  Dest.Size:=DestLen;

  zstream.zalloc:=zlibAllocMem;
  zstream.zfree:=zlibFreeMem;

  zstream.next_in:=Pointer(LongWord(Source.Memory));
  zstream.avail_in:=SourceLen;
  zstream.next_out:=Dest.Memory;
  zstream.avail_out:=DestLen;
  InflateInit_(zstream,ZLIB_VERSION,SizeOf(TZStreamRec));
  try
    inflate(zstream,Z_NO_FLUSH);
  finally
    inflateEnd(zstream);
  end;
end;

Function PAC_Ana_Res(str_File_Path:String):Boolean; //Res Ana
Var
hdl_File_Handle:Integer;
Head:Pack_Head;
TempBuffer:Pbyte;
Int_vnCounter,i:Integer;
Out_Handle:Integer;
entries:array of Pack_Entries;
Compress,decompress:Tmemorystream;
decompress_length:DWORD;

begin
Result:=True;
try
hdl_File_Handle:=fileopen(str_File_Path,fmopenread);
if hdl_File_Handle=-1 then  raise Exception.Create('Can Not Open File');
FileRead(hdl_File_Handle,Head.Magic[0],$20); //文件头
TempBuffer:=sysgetmem(HEAD.entries_length);
FileRead(hdl_File_Handle,TempBuffer[0],HEAD.entries_length);
decompress:=Tmemorystream.Create;
Compress:=Tmemorystream.Create;
Compress.Write(TempBuffer[0],HEAD.entries_length);
compress.Position:=0;
decompress.Position:=0;
ZlibDecompressStream(compress,decompress,HEAD.decompressed_length);
compress.Destroy;
sysfreemem(TempBuffer);
decompress.Position:=0;
i:=0;
Int_vnCounter:=0;
getmem(pbyte(entries),HEAD.decompressed_length);
while i<HEAD.decompressed_length do
begin
decompress.Read(entries[Int_vnCounter].TOTALENGTH,4);
decompress.Read(entries[Int_vnCounter].ZERO,4);
decompress.Read(entries[Int_vnCounter].START_OFFSET,4);
decompress.Read(entries[Int_vnCounter].ZERO1,4);
decompress.Read(entries[Int_vnCounter].COMPRESS_FLAG,4);
decompress.Read(entries[Int_vnCounter].Length,4);
decompress.Read(entries[Int_vnCounter].DECOMPRESSED_LENGTH,4);
SETLENGTH(entries[Int_vnCounter].File_name,(entries[Int_vnCounter].TOTALENGTH-28)div 2);
decompress.Read(PBYTE(entries[Int_vnCounter].File_name)^,entries[Int_vnCounter].TOTALENGTH-28);
i:=i+entries[Int_vnCounter].TOTALENGTH;
inc(Int_vnCounter);
end;
decompress.Destroy;
//读入数据
i:=0;
Int_vnCounter:=0;
str_File_Path:=ExtractFilePath(str_File_Path)+'OUTPUT\'+ ChangeFileExt(extractfilename(str_File_Path),'') +'\';
while i<HEAD.decompressed_length do
BEGIN
ForceDirectories(ExtractFilePath(str_File_Path+ExtractFilePath(ChangeFileExt(trim(entries[Int_vnCounter].File_Name),''))));
Out_Handle:=Filecreate(str_File_Path+trim(entries[Int_vnCounter].File_Name));
if Out_Handle=-1 then  Out_Handle:=Fileopen(str_File_Path+trim(entries[Int_vnCounter].File_Name),fmopenwrite);
TempBuffer:=sysgetmem(entries[Int_vnCounter].Length);
Fileseek(hdl_File_Handle,entries[Int_vnCounter].START_OFFSET+head.DATA_OFFSET,0);
FileRead(hdl_File_Handle,TempBuffer[0],entries[Int_vnCounter].Length);
if entries[Int_vnCounter].COMPRESS_FLAG=1 then
begin
decompress:=Tmemorystream.Create;
Compress:=Tmemorystream.Create;
Compress.Write(TempBuffer[0],entries[Int_vnCounter].Length);
compress.Position:=0;
decompress.Position:=0;
ZlibDecompressStream(compress,decompress,entries[Int_vnCounter].decompressed_length);
compress.Destroy;
sysfreemem(TempBuffer);
TempBuffer:=sysgetmem(decompress.Size);
decompress.Position:=0;
decompress.Read(TempBuffer[0],decompress.Size);
entries[Int_vnCounter].Length:=decompress.Size;
decompress.Destroy;
end;
FileWrite(Out_Handle,TempBuffer[0],entries[Int_vnCounter].Length);
sysfreemem(TempBuffer);
FileClose(Out_Handle);
i:=i+entries[Int_vnCounter].TOTALENGTH;
inc(Int_vnCounter);
END;
FileClose(hdl_File_Handle);
except
Result:=False;
Exit;
end;
end;
