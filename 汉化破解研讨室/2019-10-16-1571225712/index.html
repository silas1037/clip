年初写的Quartett!的文本提取程序的下文就在此：文本插入程序。
荒废了N久的ScriptInserter终于做了 OTL

使用方法：
把编过号的*.tkn文件、ScriptInserter.exe与前面提到的文本提取程序提取出来的*.txt文件放在同一目录下（下面假设目录路径为D:\test\）。然后执行命令：
Java代码  
  1. ScriptInserter textfile.txt  

这样就会在名为new的子目录（假设是D:\test\new\）里得到目标*.tkn文件。

接下来……直接发代码吧。这代码基本上就是提取程序加了十来行代码的样子吧，基本逻辑几乎一模一样所以也没代码什么需要详细说明的。
主要的逻辑是：
0、校验相关文件是否存在，并创建一个名为new的子目录
1、校验*.tkn文件的signature，正确的话将signature也写入新的文件
2、获取总的token数，并写入新的文件
3、对每个token，
a、读入行号，并写入新文件
b、读入token类型，并写入新文件
c、读入token的值，并根据token类型来转换当前的程序状态；依据状态来决定写入原tkn的token值还是从txt读入文本写入新文件
4、结束

  28. using System;  
  29. using System.Collections.Generic;  
  30. using System.IO;  
  31. using System.Text;  
  32.   
  33. namespace FFDSystemAnalysis  
  34. {  
  35.     enum TokenType  
  36.     {  
  37.         Decimal = 0x080,  
  38.         Identifier = 0x081,  
  39.         Hexadecimal = 0x082,  
  40.         String = 0x083,  
  41.         Operator = 0x085  
  42.     }  
  43.       
  44.     enum OutputState  
  45.     {  
  46.         Searching,  
  47.         FoundText,  
  48.         FoundEqual  
  49.     }  
  50.   
  51.     sealed class ScriptInserter  
  52.     {  
  53.         private static readonly byte[ ] SIGNATURE = {  
  54.             ( byte )0x54, ( byte )0x4F, ( byte )0x4B, ( byte )0x45,  
  55.             ( byte )0x4E, ( byte )0x53, ( byte )0x45, ( byte )0x54,  
  56.             ( byte )0x64, ( byte )0x0,  ( byte )0x0,  ( byte )0x0  
  57.         };  
  58.   
  59.         static void Main( string[ ] args ) {  
  60.             string infile = args[ 0 ];  
  61.             if ( !infile.EndsWith( ".txt" ) ) return;  
  62.             if ( !File.Exists( infile ) ) return;  
  63.   
  64.             string srcfile = Path.GetFileNameWithoutExtension( infile ) + ".tkn";  
  65.             if ( !File.Exists( srcfile ) ) return;  
  66.             string outfile = "./new/" + Path.GetFileNameWithoutExtension( infile ).Substring( 4 ) + ".tkn";  
  67.             Directory.CreateDirectory( "./new" );  
  68.   
  69.             Encoding utf16le = new UnicodeEncoding( false, true );  
  70.             Encoding jis = Encoding.GetEncoding( 932 );  
  71.             Encoding gbk = Encoding.GetEncoding( 936 );  
  72.   
  73.             using ( BinaryReader srcreader = new BinaryReader( File.OpenRead( srcfile ), jis ) ) {  
  74.                 using ( StreamReader inreader = new StreamReader( infile, utf16le) ) {  
  75.                     using ( BinaryWriter writer = new BinaryWriter( File.Create( outfile ), gbk ) ) {  
  76.                         byte[ ] sig = srcreader.ReadBytes( SIGNATURE.Length );  
  77.                         if ( !Equals( sig, SIGNATURE ) ) {  
  78.                             Console.WriteLine( "Wrong signature" );  
  79.                             return;  
  80.                         }  
  81.       
  82.                         // write UTF-16LE BOM  
  83.                         writer.Write( SIGNATURE );  
  84.       
  85.                         // process each token  
  86.                         int lineNum = 1;  
  87.                         OutputState state = OutputState.Searching;  
  88.                         int tokenCount = srcreader.ReadInt32( );  
  89.                         writer.Write( tokenCount );  
  90.                           
  91.                         for ( int tokenNum = 0; tokenNum < tokenCount; ++tokenNum ) {  
  92.                             // get line number  
  93.                             lineNum = srcreader.ReadInt32( );  
  94.                             writer.Write( lineNum );  
  95.       
  96.                             // get tokenType  
  97.                             TokenType tokenType = ( TokenType ) ( srcreader.ReadByte( ) & 0x0FF );  
  98.                             writer.Write( ( byte ) tokenType );  
  99.       
  100.                             // get token value  
  101.                             string tokenString = ReadCString( srcreader );  
  102.       
  103.                             switch ( tokenType ) {  
  104.                               
  105.                             case TokenType.Identifier:  
  106.                                 if ( tokenString.Equals("text") ) {  
  107.                                     state = OutputState.FoundText;  
  108.                                 } else {  
  109.                                     state = OutputState.Searching;  
  110.                                 }  
  111.                                 writer.Write( gbk.GetBytes( tokenString ) );  
  112.                                 writer.Write( ( byte ) 0 );  
  113.                                 break;  
  114.       
  115.                             case TokenType.String:  
  116.                                 if (OutputState.FoundEqual == state)  
  117.                                 {  
  118.                                     int inLinenum = Convert.ToInt32( inreader.ReadLine( ) );  
  119.                                     if ( lineNum != inLinenum ) throw new Exception( "Line number doesn't match." );  
  120.                                     string inText = inreader.ReadLine( );  
  121.                                     writer.Write( gbk.GetBytes( inText ) );  
  122.                                     writer.Write( ( byte ) 0 );  
  123.                                     state = OutputState.Searching;  
  124.                                 } else {  
  125.                                     writer.Write( gbk.GetBytes( tokenString ) );  
  126.                                     writer.Write( ( byte ) 0 );  
  127.                                 }  
  128.                                 break;  
  129.       
  130.                             case TokenType.Operator:  
  131.                                 if ((OutputState.FoundText == state)  
  132.                                     && tokenString.Equals("=")) {  
  133.                                     state = OutputState.FoundEqual;  
  134.                                 } else {  
  135.                                     state = OutputState.Searching;  
  136.                                 }  
  137.                                 writer.Write( gbk.GetBytes( tokenString ) );  
  138.                                 writer.Write( ( byte ) 0 );  
  139.                                 break;  
  140.                               
  141.                             case TokenType.Decimal:  
  142.                             case TokenType.Hexadecimal:  
  143.                                 writer.Write( gbk.GetBytes( tokenString ) );  
  144.                                 writer.Write( ( byte ) 0 );  
  145.                                 break;  
  146.                                   
  147.                             default:  
  148.                                 Console.WriteLine( "Unexpected token type {0} at 0x{1}.",  
  149.                                     tokenType.ToString( "X" ),  
  150.                                     srcreader.BaseStream.Position.ToString( "X" ) );  
  151.                                 return;  
  152.                             } // switch tokenType  
  153.                         } // for  
  154.                     }  
  155.                 }  
  156.             }  
  157.         }  
  158.   
  159.         static bool Equals( byte[ ] a, byte[ ] b ) {  
  160.             int len = a.Length;  
  161.             if ( len != b.Length ) return false;  
  162.             for ( int i = 0; i < len; i++ ) {  
  163.                 if ( a[ i ] != b[ i ] ) return false;  
  164.             }  
  165.             return true;  
  166.         }  
  167.   
  168.         static string ReadCString( BinaryReader srcreader ) {  
  169.             StringBuilder builder = new StringBuilder( );  
  170.             char c = '\0';  
  171.   
  172.             while ( ( c = srcreader.ReadChar( ) ) != '\0' ) {  
  173.                 builder.Append( c );  
  174.             }  
  175.   
  176.             return builder.ToString( );  
  177.         }  
  178.     }  
  179. }
