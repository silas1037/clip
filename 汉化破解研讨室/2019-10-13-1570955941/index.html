static    void    nnnstr(uchar *in) {
    int    p;

    p=0;
    while (in[p]) {
        if (in[p]<' ') {
            in[p] = '\0';
            break;
        }
        p++;
    }
}

static    void    mk_path(uchar *to, uchar *in, uchar *atama) {
    uchar    drive[30], dir[80], fname[80], ext[50];

    if (atama[0]==0) {
        strcpy(to, in);
        return;
    }                    //  偁偨傑側偟

    strcpy(to, atama);
    if (to[strlen(to)-1]!='\\') {
        strcat(to, "\\");
    }
    _splitpath(in, drive, dir, fname, ext);

    strcat(to, fname);
    strcat(to, ext);
}
void    fileadd(uchar *fn) {
    static    uchar    file[_MAX_PATH];
    FILE    *pp;
    uchar    drive[30], dir[80], fname[80], ext[50];

    mk_path(file, fn, from);

    _splitpath(fn, drive, dir, fname, ext);
    strcat(fname, "          ");
    strcat(ext, "          ");

    strupr(fname);    strupr(ext);

    memcpy(filepos[pps].fname, fname, 8);
    memcpy(&filepos[pps].fname[8], &ext[1], 3);
    filepos[pps].fname[11] = '\0';
    filepos[pps].pos = pps_len;

    if (NULL==(pp=fopen(fn, "rb"))) {
        printf("FILE [%s]偑傒偮偐傝傑偣傫\n", fn);
        exit(1);
    }
    pps_len += filelength(fileno(pp));
    fclose(pp);
    pps++;

}

void    main(int argc, uchar *argv[]) {
    static    uchar    outfile[_MAX_PATH];
    uint    i, pos, ecc;
    long    llp;
    uchar    buff[100];
    FILE    *fp;

    ecc = 0;
    memset(filepos, 0, sizeof(filepos));
    usage();

    memset(from, 0, sizeof(from));
    memset(to, 0, sizeof(to));

    if (NULL==(fp=fopen(argv[1], "rt"))) {
        exit(1);
    }
    for (;;) {
        if (NULL==fgets(buff, 100, fp))    break;
        strupr(buff);    nnnstr(buff);

        if (buff[0]=='>') {

            printf("僼傽僀儖弌椡愭僨傿儗僋僩儕[%s]\n", to);

            printf("僼傽僀儖強摼尦僨傿儗僋僩儕[%s]\n", from);
