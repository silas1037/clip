Procedure Lz77_HatsuneTakumi_Compress_Stream(Input:Tmemorystream;Output:Tmemorystream;const Length:Dword);
var
C:Byte;
begin
C:=00;
Input.Position:=0;
Output.Position:=0;
while Input.Position<Length  do
begin
Output.Write(C,1);
Output.CopyFrom(Input,8);
end;
C:=$C0;
Output.Write(C,1);
C:=($00 shr 4)and $f0;
Output.Write(C,1);
C:=00;
Output.Write(C,1);
end;

Procedure Lz77_HatsuneTakumi_Stream(Input:Tmemorystream;Output:Tmemorystream;const Length:Dword);
label loop,loop2,finish,con;
var
ebx,edx,esi,edi,ebp,MyPosition:Dword;
dl:Byte;
c:Integer;
begin
//初始化常量
ebx:=1;
//保证总是在开始处
input.Position:=0;
output.Position:=0;
loop:
while ebx>0 do
begin
if input.Position>=Length then Goto Finish;
Input.Read(dl,1);
if dl=0 then
begin
//如果标识符为00,那么后面8个字节均为未压缩数据
output.CopyFrom(Input,8);
//下一轮
goto loop;
end;
//否则则是存在压缩过的数据
edx:=dl;
edx:=(edx shl $18)OR$800000;
ebp:=edx;
//最常见的解压主体
loop2:
while True do
begin
if input.Position>=Length then Goto Finish;
if integer(ebp)>=0 then
begin
//js 注意类型转换 否则位标志不起作用
MyPosition:=0;
while (integer(ebp)>0)AND(MyPosition<7) do
begin
ebp:=ebp shl 1;
output.CopyFrom(Input,1);
INC(MyPosition);
end;
end;
begin
ebp:=ebp shl 1;
//尽量用Goto而不是Break
if ebp=0 then goto loop;
Input.Read(dl,1);
esi:=output.Position;
edi:=dl;
edx:=dl;
edx:=(edx and $0f0) shl 4;
Input.Read(dl,1);
esi:=esi-(edx+dl);
//位置为0,既没有压缩或压缩是当前位置,既是结束标志
if esi=Input.Position then Goto Finish;
edi:=edi and $0f;
if edi=0 then
begin
Input.Read(dl,1);
edi:=dl+$10;
end else
//太麻烦了,还是用汇编写一个算了
//算了 老老实实的写完吧 免得又出现bug
begin
if edi=1 then
begin
//注意此处的位置
MyPosition:=output.Position;
output.Seek(esi,soFromBeginning);
output.Read(dl,1);
output.Seek(MyPosition,soFromBeginning);
output.Write(dl,1);
MyPosition:=output.Position;
output.Seek(esi+1,soFromBeginning);
output.Read(dl,1);
output.Seek(MyPosition,soFromBeginning);
output.Write(dl,1);
Goto loop2;
end;
end;
//复制压缩部分
MyPosition:=output.Position;
output.Seek(esi,soFromBeginning);
output.Read(dl,1);
output.Seek(MyPosition,soFromBeginning);
output.Write(dl,1);
inc(esi);
for c := 0 to edi - 1 do
begin
MyPosition:=output.Position;
output.Seek(esi,soFromBeginning);
output.Read(dl,1);
output.Seek(MyPosition,soFromBeginning);
output.Write(dl,1);
inc(esi);
end;
end;
end;
finish:
dec(ebx);
end;
end;
