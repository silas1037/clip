procedure ZlibDecompressStream(Source, Dest: TMemoryStream; DestLen: LongWord);
var
  zstream: TZStreamRec;
  SourceLen: LongWord;
begin
  FillChar(zstream, SizeOf(TZStreamRec), 0);
  SourceLen := Source.Size;
  Source.Position := 0;
  Dest.Size := DestLen;

  zstream.zalloc := zlibAllocMem;
  zstream.zfree := zlibFreeMem;

  zstream.next_in := Pointer(LongWord(Source.Memory));
  zstream.avail_in := SourceLen;
  zstream.next_out := Dest.Memory;
  zstream.avail_out := DestLen;
  InflateInit_(zstream, ZLIB_VERSION, SizeOf(TZStreamRec));
  try
    inflate(zstream, Z_NO_FLUSH);
  finally
    inflateEnd(zstream);
  end;
end;

Procedure Decode(Buffer: Pbyte; Length: DWORD; decode_type: byte);
VAR
  AX, CX: DWORD;
  i: Integer;
  TEMP: DWORD;
begin
  for i := 0 to Length - 1 do
  BEGIN
    if Buffer = $00 then
      Break;
    TEMP := $CCCCCCCC; // Pdword;
    AX := Buffer; // byte
    AX := AX and $F0;
    AX := AX shr 4;
    AX := AX or $80000000;
    Move(AX, TEMP, 1);
    CX := Buffer; // byte
    CX := CX and $F;
    CX := CX shl 4;
    Move(CX, Pbyte(@TEMP)[1], 1);
    AX := Pbyte(@TEMP)[0];
    CX := Pbyte(@TEMP)[1];
    AX := AX xor CX;
    Buffer := AX;
  END;
end;

function Get_Pack_Header(input: Pointer; input_length: Cardinal): TPack_Header;
  stdcall;
var
  head: Pack_Head;
begin
  if input_length <> $4 then
  begin
    Result.Err_Code.Num := Unknown_Pack;
    Result.Err_Code.Info := Unknown_Pack_S;
    Exit;
  end;
  Move(input^, head.ENTRIES_COUNT, $4);
  FillMemory(@Result.Magic, SizeOf(TPack_Header), 0);
  Result.ENTRIES_COUNT := head.ENTRIES_COUNT;
  Result.Entries_Offset := -($4 + head.ENTRIES_COUNT * $50);
  Result.Entries_Origin := 2;
  Result.Entries_Length_O := head.ENTRIES_COUNT * $50;
  Result.Entries_Length := Result.Entries_Length_O;
  Result.Data_Offset := 0;
  SetLength(Result.Entries_Array, 0);
end;

function Get_Pack_Entries(input: Pointer; input_length: Cardinal;
  head: TPack_Header; EntriesArray: TAdditional_Entries_Array): TMy_Entries;
  stdcall;
var
  Entries: array of Pack_Entries;
  i: Integer;
  FileName: AnsiString;
begin
  SetLength(FileName, $40);
  if input_length <> head.Entries_Length_O then
  begin
    Result.Entries := nil;
    Result.Entries_Num := 0;
    Exit;
  end;
  SetLength(Entries, head.ENTRIES_COUNT);
  SetLength(Result.Entries, head.ENTRIES_COUNT);
  Move(input^, Entries[0].file_name[1], head.Entries_Length_O);
  for i := 0 to head.ENTRIES_COUNT - 1 do
  begin
    Decode(@Entries.file_name[1], $40, 0);
    Decode(@Entries.start_offset, $4, 0);
    Decode(@Entries.Length, $4, 0);
    Decode(@Entries.Decompressed_Length, $4, 0);
    Decode(@Entries.reserved, $4, 0);
    Result.Entries.Offset := Entries.start_offset;
    Result.Entries.Length := Entries.Length;
    Result.Entries.Decompressed_Length := Entries.Decompressed_Length;
    Move(Entries.file_name[1], Pbyte(FileName)^, $40);
    Result.Entries.Name := StrAlloc($40 + 1);
    FillMemory(@Result.Entries.Name[0], $40 + 1, 0);
    Move(Pbyte(string(FileName))^, Pbyte(Result.Entries.Name)^, $40);
  end;
end;

  begin
    Result.Err_Code.Num := Unknown_Pack;
    Result.Err_Code.Info := Unknown_Pack_S;
    Exit;
  end;
  move(input^, head.Magic[0], $20);
  FillMemory(@Result.Magic, SizeOf(TPack_Header), 0);
  setlength(Magic, 8);
  move(head.Magic[0], Pbyte(Magic)^, 8);
  Result.Magic := pchar(trim(Magic));
  Result.ENTRIES_COUNT := head.ENTRIES_COUNT;
  Result.Entries_Offset := $20;
  Result.Entries_Length_O := head.ENTRIES_COUNT * SizeOf(Pack_Entries);
  Result.Entries_Length := Result.Entries_Length_O;
  Result.Data_Offset := Result.Entries_Offset + Result.Entries_Length_O;
  setlength(Result.Entries_Array, 0);
end;

function Get_Pack_Entries(input: Pointer; input_length: Cardinal;
  head: TPack_Header; EntriesArray: TAdditional_Entries_Array): TMy_Entries;
  stdcall;
var
  Entries: array of Pack_Entries;
  i: Integer;
  Filename: Ansistring;
begin
  Decode(input, input_length, 0);
  setlength(Filename, $110 - $C);
  if input_length <> head.Entries_Length_O then
  begin
    Result.Entries := nil;
    Result.Entries_Num := 0;
    Exit;
  end;
  setlength(Entries, head.ENTRIES_COUNT);
  setlength(Result.Entries, head.ENTRIES_COUNT);
  move(input^, Entries[0].file_name[1], head.Entries_Length_O);
  for i := 0 to head.ENTRIES_COUNT - 1 do
  begin
    Filename := '';
    Get_Real_FileName(Entries.file_name);
    setlength(Filename, $110 - $C);
    Result.Entries.Offset := Entries.start_offset;
    Result.Entries.Length := Entries.Length;
    Result.Entries.Decompressed_Length := Entries.Decompressed_Length;
    move(Entries.file_name[1], Pbyte(Filename)^, $110 - $C);
    Result.Entries.Name := StrAlloc($110 - $C);
    FillMemory(@Result.Entries.Name[0], $110 - $C * 2, 0);
    move(Pbyte(Format('%s', [Filename]))^, Pbyte(Result.Entries.Name)^,
      Length(Filename));
  end;
end;

// 除非数据是无任何变换的明文,否则均不允许直接使用原Buffer来传递数据.
function Get_Pack_Data(input: Pointer; input_length: Cardinal;
  head: TPack_Header; Entry_Info: TPack_Entries): TData_Result; stdcall;
var
  Com, decom: TMemoryStream;
  Buffer: Pbyte;
begin
  if Entry_Info.Decompressed_Length <> 0 then
  begin
    Decode(input, $19, 0);
    Buffer := SysGetMem(Entry_Info.Decompressed_Length);
    decom := TMemoryStream.create;
    Com := TMemoryStream.create;
    Com.Write(input^, input_length);
    Com.Position := 0;
    ZlibDecompressStream(Com, decom, Entry_Info.Decompressed_Length);
    Com.Destroy;
    if decom.Size <> Entry_Info.Decompressed_Length then
    begin
      Result.Err_Code.Num := Compress_Error;
      Result.Err_Code.Info := Compress_Error_S;
      Exit;
    end;
    decom.Position := 0;
    decom.Read(Buffer[0], decom.Size);
    Result.Data := Buffer;
    Result.Length := decom.Size;
    decom.Destroy;
    Result.Err_Code.Num := 0;
    Result.Err_Code.Info := nil;
    Exit;
  end
  else
  begin
    Result.Data := input;
    Result.Length := input_length;
    Result.Err_Code.Num := 0;
    Result.Err_Code.Info := nil;
    Exit;
  end;
end;
