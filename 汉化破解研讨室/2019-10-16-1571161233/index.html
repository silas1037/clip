GalgameKIDRemember11解包

r11 的封包是 afs 格式的，封包没有压缩和加密，在封包头部有文件的偏移和大小，封包尾部有各文件的文件名以及其他一些信息。

    所有封包中含有以下类型文件：

  • bip: lzss 压缩或未压缩的多 png 组合文件
  • t2p: lzss 压缩的 tim2 文件，tim2 可转为 bmp 文件
  • adx: 原始 ogg 或 wav 声音文件

    • afs 相关结构

    1typedef struct
    2{
    3    char       Magic[4];
    4    int        FileCount;
    5    AFS_ENTRY *Entries;
    6} AFS_HEADER;
    7 
    8typedef struct      // 紧跟文件头的索引
    9{
    10    int Offset;
    11    int Length;
    12} AFS_ENTRY;
    13 
    14typedef struct      // 封包尾部的索引信息
    15{
    16    char FileName[32];
    17    char UnknownBytes[16];
    18} AFS_INDEX;

    t2p 相关结构

    解压后得到 tim2 文件。
    tim2 文件结构如下，与 bmp 类似的两个数据结构，32 位色的情况下后面跟随 4 字节的像素信息。

    1typedef struct
    2{
    3    char           FileId[4]; // ID of the File (must be 'T', 'I', 'M' and '2')
    4    unsigned char  FormatVersion; // Version number of the format
    5    unsigned char  FormatId; // ID of the format
    6    unsigned short Pictures; // Number of picture data
    7    char           Reserved[8]; // Padding (must be 0x00)
    8} TIM2_FILEHEADER;
    9 
    10typedef struct
    11{
    12    unsigned long  TotalSize; // Total size of the picture data in unsigned chars
    13    unsigned long  ClutSize; // CLUT data size in unsigned chars
    14    unsigned long  ImageSize; // Image data size in unsigned chars
    15    unsigned short HeaderSize; // Header size in unsigned chars
    16    unsigned short ClutColors; // Total color number in CLUT data
    17    unsigned char  PictFormat; // ID of the picture format (must be 0)
    18    unsigned char  MipMapTextures;// Number of MIPMAP texture
    19    unsigned char  ClutType; // Type of the CLUT data
    20    unsigned char  ImageType; // Type of the Image data
    21    unsigned short ImageWidth; // Width of the picture
    22    unsigned short ImageHeight; // Height of the picture
    23    unsigned char  GsTex0[8]; // Data for GS TEX0 register
    24    unsigned char  GsTex1[8]; // Data for GS TEX1 register
    25    unsigned long  GsRegs; // Data for GS TEXA, FBA, PABE register
    26    unsigned long  GsTexClut; // Data for GS TEXCLUT register
    27} TIM2_PICTUREHEADER;
    28 
    29// 其中 ClutType 具体取值如下：
    30// 01   16 位直接颜色文件
    31// 02   24 位直接颜色文件
    32// 03   32 位直接颜色文件
    33// 04   4 位索引文件（16色）
    34// 05   8 位索引文件（256色）
    35 
    36// ImageType 取值如下：
    37// 00 无调色板
    38// 索引颜色文件时
    39// 高四位
    40//      0 采用 CSM1 调色板
    41//      1 采用 CSM2 调色板
    42// 低四位
    43//      1 16位色调色板
    44//      2 24位色调色板
    45//      3 32位色调色板

    bip 相关结构

    该结构跟随在 “EMUARC__” 标志之后，之后紧跟 png 图片数据

    1typedef struct
    2{
    3    char   Magic[8];        // "PNGFILE2"
    4    int    PartWidth;
    5    int    PartHeight;
    6    short  FullWidth;
    7    short  FullWidthThumb;
    8    short  FullHeight;
    9    short  FullHeightThumb;
    10    int    PartSize;        // header + pngfile
    11    char   FileNameWithDirectory[64];
    12    char   Unknown[16];
    13    int    PicOffsetX;
    14    int    PicOffsetY;
    15    int    PicWidth;
    16    int    PicHeight;
    17} PNG_PACK_TYPE2;

    压缩后的 bip 经解压可得到

    1typedef struct
    2{
    3    TIM2_FILEHEADER    Tim2FileHeader;
    4    TIM2_PICTUREHEADER Tim2PicHeader;
    5    char               Magic[8];        // "PNGFILE3"
    6    int                PartWidth;
    7    int                PartHeight;
    8    short              FullWidth;
    9    short              FullWidthThumb;
    10    short              FullHeight;
    11    short              FullHeightThumb;
    12    int                PartSize;        // header from magic + pngfile
    13    char               FileNameWithDirectory[64];
    14    char               Unknown[16];
    15    int                PicOffsetX;
    16    int                PicOffsetY;
    17    int                PicWidth;
    18    int                PicHeight;
    19} PNG_PACK_TYPE3;

    该结构也是一个跟随一个出现，但是结构的起始地址要对齐16字节的文件偏移。
    该数据结构之后同样跟随 png 图片数据。
