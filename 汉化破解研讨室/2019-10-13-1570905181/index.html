    文本提取程序 （以C编写）

    程序样貌




    文本提取程序+秋之回忆6：三角波动文本下载地址：

    http://www.megaupload.com/?d=X03F6DYY

    文本提取程序代码讲解：

    _______________________________________________________________________________________________________________________________
    //printf,fopen,fclose.fputc.fgetc
    #include <stdlib.h>
    //exit
    #include <string.h>
    //strcmp,strlen

    typedef unsigned char byte;//定义字节类型

    int main(int argc,char** argv){
        if(argc!=3){//检测参数个数，不合要求则打印参数格式并退出
            printf("usage:%s file_in file_out\n",argv[0]);
            return 1;
        }
        FILE* f=fopen(argv[1],"rb");//以二进制只读形式打开第一个参数指定的文件
        if(!f){//如果打开失败，打印错误信息并退出
            printf("file %s open failed\n",argv[1]);
            return 1;
        }
        byte key[]={0xff,0xff,0xf8,0x02,0x24,0x04};//bin文件中用于定位文本的特征字节串
        byte buf[102400];//缓冲区，用于存放文件内容
        int flength=0;//记录文件长度
        while(1){//将文件全部读入缓冲区
            byte c=fgetc(f);
            if(feof(f)){
                break;
            }
            buf[flength++]=c;
        }
        fclose(f);//关闭文件
    //    puts("file loaded");
        int i,j;
        for(i=0;i<flength;i++){//搜索特征字节串
            for(j=0;j<sizeof(key);j++){
                if(buf[i+j]!=key[j]){
                    break;
                }
            }
            if(j==sizeof(key)){
                break;
            }
        }
        typedef short uint16;
        uint16 text_addr;
        if(i==flength){//如果没有找到特征字节串，打印信息并退出
            puts("text not found");
            exit(2);
        }else{//否则获取文本在文件中的偏移地址
            text_addr=*(uint16*)(buf+i+j)+0x14;
    //        printf("text found at 0x%X\n",(int)text_addr);
        }
        int text_end=flength;
        for(i=text_addr;i<flength;i++){//查找文本结束位置
            if(strncmp((char*)buf+i,"%K%P",4)==0){
                text_end=i;
            }
        }
        char* replace[]={//需要替换的控制字符串
            "%K%P",
            "%K%p",
            "%D",
            "%CFF88",
            "%S03",
            "%F",
            "%t00",
            "%t02",
            "%FE",
            "%N",
            "%",
            ""
        };
        FILE* fo=fopen(argv[2],"wb");//创建存放输出文本的文本文件
        if(!fo){//失败处理
            printf("file %s open failed\n",argv[2]);
            return 3;
        }
        int found_replace;
        for(i=text_addr;i<text_end;i++){//替换控制字符串，输出结果
            found_replace=0;
            for(j=0;strlen(replace[j])>0;j++){
                if(strncmp((char*)buf+i,replace[j],strlen(replace[j]))==0){
                    i+=strlen(replace[j]);
                    fputc('\r',fo);
                    fputc('\n',fo);
                    found_replace=1;
                    break;
                }
            }
            if(found_replace){
                continue;
            }
            fputc(buf,fo);
        }
        fclose(fo);//关闭输出文件
    //    puts("done");
        return 0;//正常退出
    }

    ______________________________________________________________________________________________________________________________
