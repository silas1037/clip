Function Seek_Data(Hdl_File_Handle: Integer; DIR: DIR_ENTRIES;
  Extract_Path: Ansistring): Boolean;
var
  DIR_Array: ARRAY OF DIR_ENTRIES;
  entries: Array of Pack_Entries;
  Int_vnCounter, out_handle: Integer;
  File_Counter, Dir_Counter: DWORD;
  TempBuffer: pbyte;
begin
  FileSeek(Hdl_File_Handle, DIR.FILE_ENTRIES_OFFSET, 0);
  FileRead(Hdl_File_Handle, Dir_Counter, $4);
  if Dir_Counter = 0 then
  begin
    FileRead(Hdl_File_Handle, File_Counter, $4);
    Setlength(entries, File_Counter);
    FileRead(Hdl_File_Handle, entries[0].FILE_NAME[1], File_Counter * $28);
    for Int_vnCounter := 0 to File_Counter - 1 do
    begin
      if not DirectoryExists(ExtractFilePath
          (Extract_Path + entries[Int_vnCounter].FILE_NAME)) then
        ForceDirectories(ExtractFilePath(Extract_Path + entries[Int_vnCounter]
              .FILE_NAME));
      out_handle := filecreate(Extract_Path + entries[Int_vnCounter].FILE_NAME);
      if out_handle = -1 then
        out_handle := fileopen(Extract_Path + entries[Int_vnCounter].FILE_NAME,
          fmopenwrite);
      TempBuffer := sysgetmem(entries[Int_vnCounter].Length);
      FileSeek(Hdl_File_Handle,
        entries[Int_vnCounter].Start_Offset + DIR.DATA_START_OFFSET, 0);
      FileRead(Hdl_File_Handle, TempBuffer[0], entries[Int_vnCounter].Length);
      filewrite(out_handle, TempBuffer[0], entries[Int_vnCounter].Length);
      sysfreemem(TempBuffer);
      fileclose(out_handle);
    end;
  end
  else
  begin
    Setlength(DIR_Array, Dir_Counter);
    FileRead(Hdl_File_Handle, DIR_Array[0].DIR_NAME[1], $28 * Dir_Counter);
    for Int_vnCounter := 0 to Dir_Counter - 1 do
    begin
      Seek_Data(Hdl_File_Handle, DIR_Array[Int_vnCounter],
        Extract_Path + Trim(DIR_Array[Int_vnCounter].DIR_NAME) + '\');
    end;
  end;
end;

Function PAC_Ana_Scr(str_File_Path: String): Boolean;
var

  Compressed_Buffer: pbyte;
  Compr: TmemoryStream;
  File_Size: Integer;
  PNG: TPngImage;
begin
  Result := True;
  try
    // 图像没有文件头 全部是同样的加密方式
    Hdl_File_Handle := fileopen(str_File_Path, fmopenread);
    if Hdl_File_Handle = -1 then
      raise Exception.Create('Can Not Open File');
    File_Size := FileSeek(Hdl_File_Handle, 0, 2);
    Compressed_Buffer := sysgetmem(File_Size);
    FileSeek(Hdl_File_Handle, 0, 0);
    FileRead(Hdl_File_Handle, Compressed_Buffer[0], File_Size);
    fileclose(Hdl_File_Handle);
    for out_handle := 0 to File_Size - 1 do
      Compressed_Buffer[out_handle] := (NOT Compressed_Buffer[out_handle]) + 1;
    Compr := TmemoryStream.Create;
    Compr.Write(Compressed_Buffer[0], File_Size);
    Compr.Position := 0;
    PNG := TPngImage.Create;
    PNG.LoadFromStream(Compr);
    PNG.SaveToFile(str_File_Path + '.png');
    PNG.destroy;
    sysfreemem(Compressed_Buffer);
    Compr.destroy;
  except
    Result := False;
    Exit;
  end;

Procedure XOR_TABLE_CREATE(KEY: Integer);
VAR
  Edi: DWORD;
  COUNTER: Integer;
BEGIN
  for COUNTER := 0 to $6 * $68 do
  begin
    Edi := KEY;
    KEY := KEY * $10DCD;
    Edi := Edi AND $FFFF0000;
    INC(KEY);
    XOR_TABLE[COUNTER] := Edi;
    Edi := KEY;
    KEY := KEY * $10DCD;
    Edi := Edi SHR $10;
    XOR_TABLE[COUNTER] := XOR_TABLE[COUNTER] or Edi;
    INC(KEY);
  end;
END;

Procedure XOR_TABLE_Update1();
var
  EAX, EDX, EBP, EBX: DWORD;
  count: Integer;
begin
  for count := 0 to $E3 - 1 do
  begin
    EAX := XOR_TABLE[count];
    EBX := XOR_TABLE[count + 1];
    EAX := EAX AND $80000000;
    EBX := EBX AND $7FFFFFFF;
    EDX := XOR_TABLE[$18D + count];
    EAX := EAX OR EBX;
    EAX := EAX SHR 1;
    EBX := EBX OR $FFFFFFFE;
    EAX := EAX XOR EDX;
    EBX := EBX + 1;
    EAX := EAX XOR $9908B0DF;
    EBX := EBX AND $9908B0DF;
    EBX := EBX XOR EAX;
    XOR_TABLE[count] := EBX;
  end;
end;

Procedure XOR_TABLE_Update3();
var
  EAX, EDX, EBP, EBX: DWORD;
  count: Integer;
begin
  for count := 0 to $138 do
  begin
    EAX := XOR_TABLE[count * 2];
    EDX := XOR_TABLE[count * 2 + 1];
    EBX := EAX;
    EAX := EAX SHR $B;
    EBP := EDX;
    EDX := EDX SHR $B;
    EAX := EAX XOR EBX;
    EDX := EDX XOR EBP;
    EBX := EAX;
    EAX := EAX SHL $7;
    EBP := EDX;
    EDX := EDX SHL $7;
    EAX := EAX AND $9D2C5680;
    EDX := EDX AND $9D2C5680;
    EAX := EAX XOR EBX;
    EDX := EDX XOR EBP;
    EBX := EAX;
    EAX := EAX SHL $F;
    EBP := EDX;
    EDX := EDX SHL $F;
    EAX := EAX AND $EFC60000;
    EDX := EDX AND $EFC60000;
    EAX := EAX XOR EBX;
    EDX := EDX XOR EBP;
    EBX := EAX;
    EAX := EAX SHR $12;
    EBP := EDX;
    EDX := EDX SHR $12;
    EAX := EAX XOR EBX;
    EDX := EDX XOR EBP;
    XOR_TABLE_New[count * 2] := EAX;
    XOR_TABLE_New[count * 2 + 1] := EDX;
  end;
end;

function MakeFileList(Path, FileExt: string): TStringList;
var
  sch: TSearchrec;
begin
  Result := TStringList.Create;
  if rightStr(Trim(Path), 1) <> '\' then
    Path := Trim(Path) + '\'
  else
    Path := Trim(Path);
  if not DirectoryExists(Path) then
  begin
    Result.Clear;
    Exit;
  end;
  if FindFirst(Path + '*', faAnyfile, sch) = 0 then
  begin
    repeat
      Application.ProcessMessages;
      if ((sch.Name = '.') or (sch.Name = '..')) then
        Continue;
      if DirectoryExists(Path + sch.Name) then
      begin
        Result.AddStrings(MakeFileList(Path + sch.Name, FileExt));
      end
      else
      begin
        if (UpperCase(extractfileext(Path + sch.Name)) = UpperCase(FileExt)) or
          (FileExt = '.*') then
          Result.Add(Path + sch.Name);
      end;
    until FindNext(sch) <> 0;
    SysUtils.FindClose(sch);
  end;
end;

Function PAC_Ana_imf(str_File_Path: String): Boolean;
var
  Hdl_File_Handle, out_handle: Integer;
  Compressed_Buffer: pbyte;
  Compr: TmemoryStream;
  File_Size: Integer;
begin
  Result := True;
  try
    // 图像没有文件头 全部是同样的加密方式
    Hdl_File_Handle := fileopen(str_File_Path, fmopenread);
    if Hdl_File_Handle = -1 then
      raise Exception.Create('Can Not Open File');
    File_Size := FileSeek(Hdl_File_Handle, 0, 2);
    Compressed_Buffer := sysgetmem(File_Size);
    FileSeek(Hdl_File_Handle, 0, 0);
    FileRead(Hdl_File_Handle, Compressed_Buffer[0], File_Size);
    fileclose(Hdl_File_Handle);
    for out_handle := 0 to File_Size - 1 do
      Compressed_Buffer[out_handle] := (NOT Compressed_Buffer[out_handle]) + 1;
    Compr := TmemoryStream.Create;
    Compr.Write(Compressed_Buffer[0], File_Size);
    Compr.Position := 0;
    Compr.SaveToFile(str_File_Path + '.mpg');
    sysfreemem(Compressed_Buffer);
    Compr.destroy;
  except
    Result := False;
    Exit;
  end;
end;
