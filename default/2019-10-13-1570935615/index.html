 【高手入】请问这是什么压缩算法

Copy code
#define LEFT_CHILD    0
#define RIGHT_CHILD    1

    BYTE node[2][256];
    BYTE child_node[256];
    unsigned int curbyte = 0;
    unsigned int act_len = 0;

    while (curbyte < total_data_length) {
        unsigned int count, index;

        for (i = 0; i < 256; i++)
            node[LEFT_CHILD][i] = i;

        /* 构造树，采用位置编码. 对于不连续的位置，位置标识符+0x80表示；
         * 对于连续的位置，只有第一个+0x80，后面跟一个连续的位置计数，
         * 再后面直接跟数据，而无需位置标识符.
         */
        for (i = 0; i < 256; ) {
            count = hash[curbyte++];

            if (count > 127) {
                unsigned int step;
                step = count - 127;
                i += step;
                count = 0;
            }

            if (i > 255)
                break;

            count++;
            for (unsigned int k = 0; k < count; k++) {
                node[LEFT_CHILD][i] = hash[curbyte++];
                if (node[LEFT_CHILD][i] != i) {
                    node[RIGHT_CHILD][i] = hash[curbyte++];
                }
                i++;
            }
        }
        /* 获得压缩长度 */
        if (hash_index_header->flag & 1) {
            count = *(u16 *)&hash[curbyte];
            curbyte += 2;
        } else {
            count = *(u32 *)&hash[curbyte];
            curbyte += 4;
        }
        /* 开始解压缩 */
        unsigned int k = 0;
        while (1) {
            /* 处理每一个子树分支 */
            if (k > 0)
                index = child_node[--k];
            else {
                if (!count)
                    break;
                count--;
                index = hash[curbyte++];
            }

            /* 达到叶节点 */
            if (node[LEFT_CHILD][index] == index)
                hash_index[act_len++] = index;
            else {
                /* 先子左树后右子树 */
                child_node[k++] = node[RIGHT_CHILD][index];
                child_node[k++] = node[LEFT_CHILD][index];
            }
        }
    }


咋看一下很像huffman，但是不是。huffman是基于统计模型的编码，为了接近熵，采用的是位编码，这里显然不是；rle也不像，那么剩下的就是字典模型的算法。问题是这个算法一开始构造树的时候，已经事先把树构造好了，这点和静态huffman有点像，而一开始初始化256节点的操作很像lzw，但是lzw的字典是运行时构造的，而这里已经提前构造好了，这点又和静态huffman很像。
我这里把压缩数据和解压后的数据放在这里 方便大家比对 看看谁知道这是什么压缩算法。


目前我看的感觉是对一段原始数据进行统计，统计出没有使用的ascii字符，然后把这些没有使用的字符作为树的非叶节点，而所有出现过的字符作为叶结点，这点可以从解压算法里看出来。
你可以尝试从Ltree文件的0x29开始跟踪，你可以画出一个二叉树出来，这颗二叉树的所有叶结点组成的串就是"\x0system\\"。也就是说一个0x29就代表了"\x0system\\"。问题就是一开始构造树的那些数据是怎么出来的，或者说一开始的树是怎么建立的。


ps：实际上压缩数据是一块一块的(总长度是total_data_length），每一块一开始的数据用于构造树，然后是一个长度值，表示接下来压缩的编码长度（count），然后这一块解压完再重新初始化树，再构造新的树，然后。。。所以原始压缩的数据看起来冗余，解压后的那个数据就是解压第一块的结果。
