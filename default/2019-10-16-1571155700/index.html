爬取备份 “忧郁的弟弟” 站点 Galgame

Aug 07,2018 in Python,Game lang 繁 read (4923)

我的Galgame资源发布站 - 忧郁的弟弟 是由忧郁的弟弟提供的汉化Galgame下载站点，关于该站点介绍请访问：关于若干注意事项（新人必读） | 我的Galgame资源发布站

资源备份档分享请见：Mygalgame全站资源备份
他人抓取项目请见： Mygalgame backup

弟弟站点html结构十分规范，而且爬取特别容易。问题在于该站的资源都是用百度云进行存储，而百度云的转存与下载较为麻烦。这里我们采取抓取和转存分别进行的方法，构造备份站点。步骤如下：

  1. 对弟弟站所有页面进行抓取下载并存储。
  2. 进行百度云批量转存，使用BaiduPCS-GO进行下载操作。
  3. 使用rclone转存到GDrive以及OneDrive。
  4. 使用OneIndex进行展示~

关于“忧郁的弟弟”站点备份，Github已有类似项目，具体可参见：Beats0/www.mygalgame.com

站点爬取

目前弟弟站的百度云链接需要使用post的方式二次获取，故在获取到文章链接后，构造post表单进行获取。方法如下：

  1. import re
  2. import time
  3. import sqlite3
  4. import requests
  5. from bs4 import BeautifulSoup
  7. s = requests.Session()
  8. s.cookies.update({"switchtheme":"mygalgame2"})
  9. s.headers.update({"User-Agent":"Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/67.0.3396.99 Safari/537.36"})
  11. db = sqlite3.connect(r"f:/mygalgame.db")
  12. cur = db.cursor()
  14. cur.execute("CREATE TABLE mygalgame ("+
  15.     "raw_link   VARCHAR (255),"+
  16.     "name       VARCHAR (100),"+
  17.     "baidu_link VARCHAR (255),"+
  18.     "secret_key VARCHAR (4)   UNIQUE PRIMARY KEY,"+
  19.     "descr      TEXT"+
  20. ");"
  21. )
  23. for p in range(1,84):
  24.     page = s.get("https://www.mygalgame.com/page/{}/".format(p))
  25.     page = BeautifulSoup(page.text,"lxml")
  26.     a_list = page.select("div#article-list > div > section > div.title-article > h1 > a")
  27.     article_list = list(map(lambda x:x["href"], a_list))
  28.     print("Craw Page {}, and find {} links".format(p,len(article_list)))
  29.     for a in article_list:
  30.         page_1 = s.get(a)
  31.         if re.search("A\d{3}",page_1.text):
  32.             secret_key = re.search("A\d{3}",page_1.text).group(0)
  33.             page_2 = s.post(a,data={"e_secret_key":secret_key})
  34.             page_bs = BeautifulSoup(page_2.text,"lxml")
  35.             name_ = page_bs.select_one("div.title-article a").get_text()
  36.             baidu_link = re.search("go\.php\?url=(https?://pan.baidu.com/s/[^']+)'",page_2.text).group(1)
  37.             descr = str(page_bs.select_one("div.centent-article"))
  38.             descr = descr[:descr.find("<!-- 内容主体结束 -->")]
  39.             cur.execute("INSERT INTO `mygalgame` (`raw_link`, `name`, `baidu_link`, `secret_key`, `descr`) VALUES (?,?,?,?,?)",(a,name_,baidu_link,secret_key,descr))
  40.             print("{}: {} OK~".format(secret_key,a))
  41.             db.commit()
  42.         time.sleep(5)
  43.     print("Page {} Over~".format(p))
  44.     time.sleep(5)

百度云转存下载

使用Selenium半自动登陆并转存。代码如下，前半部分登陆手动完成，后面分享转存交由程序进行。

  1. from selenium import webdriver
  2. from selenium.webdriver.common.keys import Keys
  4. browser = webdriver.Chrome(r"d:\chromedriver.exe")
  5. browser.get("https://pan.baidu.com")  # 用户登陆（手动）
  7. uk_list = cur.execute("SELECT baidu_link,secret_key from mygalgame order by secret_key asc").fetchall()
  8. for url,key in uk_list:
  9.     try:
  10.         browser.get(url)   # 打开分享链接
  11.         browser.implicitly_wait(3)   # 等待3秒
  12.         browser.find_element_by_tag_name("input").send_keys(key,Keys.ENTER)  # 输入分享密码并回车
  13.         browser.implicitly_wait(5)
  14.         browser.find_element_by_xpath('//*[@id="shareqr"]/div[2]/div[2]/div/ul[1]/li[1]/div/span[1]').click()  # 全选分享文件
  15.         browser.find_element_by_xpath('//*[@id="bd-main"]/div/div[1]/div/div[2]/div/div/div[2]/a[1]/span/span').click()  # 点击转存按钮
  16.         browser.implicitly_wait(2)
  17.         browser.find_element_by_xpath('//*[@id="fileTreeDialog"]/div[3]').click()  # 勾选上次保存位置
  18.         browser.implicitly_wait(2)
  19.         browser.find_element_by_xpath('//*[@id="fileTreeDialog"]/div[4]/a[2]/span').click()  # 点击确认按钮
  20.         browser.implicitly_wait(2)
  21.         print("Transfer {}#{} Success~".format(url,key))
  22.     except Exception:
  23.         print("Transfer {}#{} Fail!!!!".format(url,key))

注意：

  • 估算每个Gal的体积为2G，故1T百度云盘约能转存500左右游戏。
  • 请使用开发者工具block掉一些请求，以防止页面长时间等待加快速度。例如：

而下载使用 BaiduPCS-GO。使用前面获取的BUDSS进行登陆，并使用screen挂在后台进行下载即可。（然而使用国外服务器下载还是很慢23333）

简介清洗

因为OneIndex能展示READMD.md，所以将原有html格式简介清洗为markdown格式的文件，示例格式如下：

  1. import subprocess
  2. import sqlite3
  3. import re
  5. from markdownify import markdownify as md
  6. from bs4 import BeautifulSoup
  8. process = subprocess.Popen(['rclone','lsd','GDrive:/galgame'], stdout=subprocess.PIPE)
  9. output, _  =process.communicate()
  11. output = str(output,"utf-8")
  13. outlist = output.splitlines()
  15. db = sqlite3.connect(r"f:/mygalgame.db")
  16. cur = db.cursor()
  18. temple = """
  19. # {name}
  21. > 来源于： [{source_link}]({source_link})
  23. ## 游戏截图
  25. {img_list}
  27. ## 游戏简介
  29. {introduce}
  31. ## 汉化STAFF
  33. {staff}
  36. ## 文件信息
  38. 备注： {beizhu}
  40. {note}
  42. 标签： {tag}
  43. """
  46. for i in outlist:
  47.     oid = re.search("A\d{3}",i).group(0)
  49.     print("开始处理{}".format(i))
  51.     sourcename = re.search("A\d{3}.*?$",i).group(0)
  53.     fetch_ = cur.execute("SELECT * FROM mygalgame where secret_key = ? ",[oid])
  54.     link,name,bdlink,key,descr,_ = fetch_.fetchone()
  56.     de_bs = BeautifulSoup(descr,"lxml")
  57.     img_list = de_bs.find_all("img")
  58.     img_md = "\n".join(map(lambda x:"![]({})".format(x["src"]),img_list))
  60.     introduce = staff = beizhu  = tag =  ""
  62.     main_part = de_bs.find_all("div",class_="alert-success")
  64.     if len(main_part) == 1:
  65.         i_tag = main_part[0]
  66.         introduce = md(str(i_tag))
  67.     elif len(main_part) > 1:
  68.         i_tag,s_tag = main_part[:2]
  69.         staff = md(str(s_tag))
  70.         introduce = md(str(i_tag))
  72.     beizhu_search = re.search('备注：<span style="color: #ff0000;">(.+?)</span>',descr)
  73.     if beizhu_search:
  74.         beizhu = beizhu_search.group(1)
  76.     tag_list_raw = de_bs.find_all("a",rel="tag")
  77.     if len(tag_list_raw) > 0:
  78.         tag_list = map(lambda x:x.get_text(strip=True),tag_list_raw)
  79.         tag = ", ".join(tag_list)
  82.     note = md(str(de_bs.find("div",class_="alert-info"))).replace("任何解压相关问题请看网站顶部解压必读，不看而发问直接小黑屋","").strip()
  84.     md_data = temple.format(name=name,source_link=link,img_list=img_md,introduce=introduce.replace("\n","\n\n"),
  85.     staff=staff.replace("\n","\n\n"),note=note.replace("\n","\n\n"),beizhu=beizhu,tag=tag)
  86.     print("生成的Markdown格式如下",md_data)
  88.     with open(r"F:\Temp\README.md","w",encoding="utf-8") as f:
  89.         f.write(md_data)
  90.         print("写入临时文件完成")
  92.     process = subprocess.Popen(['rclone','copy',r"F:\Temp\README.md",'GDrive:/galgame/{}'.format(sourcename)], stdout=subprocess.PIPE)
  93.     process.communicate()
  94.     print("上传 README.md 完成")
  96.     dstname = "{}@{}".format(key,name)
  97.     process = subprocess.Popen(['rclone','moveto','GDrive:/galgame/{}'.format(sourcename),'GDrive:/mygalgame/{}'.format(dstname)], stdout=subprocess.PIPE)
  98.     process.communicate()
  99.     print("文件夹 重命名+移动 完成")
